#include <stdio.h>
int main()
{
    int ar[5];
    int i,even=0,odd=0;

    for(i=0;i<5;i++)
    {
        printf("ar[%d]: ",i);
        scanf("%d",&ar[i]);


    }
    for(i=0;i<5;i++)
    {
        if(ar[i]%2==0)
        {
            even++;
        }
        if(ar[i]%2!=0)
        {
            odd++;

        }
    }
printf("Even = %d\n",even);
printf("Odd = %d",odd);
return 0;
}
