//reverse an array
#include<stdio.h>
int main()
{
    int arr[5];
    int i,j,val;

    for(i=0;i<5;i++)
    {
        printf("arr[%d]:",i);
        scanf("%d",&arr[i]);

    }
    for(i=0;i<5;i++)
    { for(j=0;j<5-i;j++)
    {


        if(arr[j]>arr[j+1])
        {
            int val=arr[j];
            arr[j]=arr[j+1];
            arr[j+1]=val;

        }
    }

    }

    for(i=0;i<5;i++)
    {
        printf("Asc:%d\n",arr[i]);

    }

    return 0;
}
