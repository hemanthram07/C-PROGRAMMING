#include <stdio.h>
int main()
{
    int arr[5];
    int n,i,diff;

    for(i=0;i<5;i++)
    {
        printf("arr[%d]=",i);
        scanf("%d",&arr[i]);

    }

    int max=arr[0];
    int min=arr[0];

    for(i=0;i<5;i++)
    {
        if(max<arr[i])
        {
            max=arr[i];
        }
        if(min>arr[i])
        {
            min=arr[i];
        }
    }
    diff=max-min;
    printf("Max=%d\n",max);
    printf("Min=%d\n",min);
    printf("diff=%d",diff);
    return 0;
}
