#include <stdio.h>
int main()
{
int mat[10][10];
int i,j,sum=0,n;
printf("Enter the values for n");
scanf("%d",&n);

for(i=0;i<n;i++)
{
for(j=0;j<n;j++)
{
printf("Enter the values=");
scanf("%d",&mat[i][j]);
}

}
for(i=0;i<n;i++)
{
for(j=0;j<n;j++)
{
if(i==j)
{
sum+=mat[i][j];
}
}

}
printf("Sum of the diagonals: %d",sum);
return 0;
}
